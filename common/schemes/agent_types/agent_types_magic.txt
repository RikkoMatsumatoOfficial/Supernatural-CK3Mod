# Witchcraft
agent_spellweaver = {
	contribution_type = success_chance_max
	
	contribution = {
		# Base value.
		value = 0

		if = {
			limit = {
				has_trait = archmage
				is_the_choosen_one_trigger = yes
			}
			add = {
				value = 100
				desc = scheme_agent_aptitude.choosen
			}
		}
		else_if = {
			limit = {
				has_trait = archmage
				has_trait_xp = {
					trait = archmage
					track = secret_knowledge
					value > 74
				}
			}
			add = {
				value = 25
				desc = scheme_agent_aptitude.magic_user
			}
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_trait = archmage
						has_trait_xp = {
						trait = archmage
						track = secret_knowledge
						value > 49
						}
					}
					has_trait = yaga
				}
				
			}
			add = {
				value = 20
				desc = scheme_agent_aptitude.magic_user
			}
		}
		else_if = {
			limit = {
				has_trait = archmage
				has_trait_xp = {
					trait = archmage
					track = secret_knowledge
					value > 24
				}
			}
			add = {
				value = 15
				desc = scheme_agent_aptitude.magic_user
			}
		}
		else_if = {
			limit = {
				is_true_mage = yes
			}
			add = {
				value = 10
				desc = scheme_agent_aptitude.magic_user
			}
		}
		else_if = {
			limit = {
				has_trait = true_witch
			}
			add = {
				value = 5
				desc = scheme_agent_aptitude.trait.true_witch
			}
		}
		else_if = {
			limit = {
				is_a_demon_trigger = yes
			}
			add = {
				value = 5
				desc = scheme_agent_aptitude.trait.demon
			}
		}
		else_if = {
			limit = {
				is_witch_trigger = yes
			}
			add = {
				value = 1
				desc = scheme_agent_aptitude.trait.witch
			}
		}
		else_if = {
			limit = {
				has_trait = lifestyle_mystic
			}
			add = {
				value = 1
				desc = scheme_agent_aptitude.trait.lifestyle_mystic
			}
		}
		if = {
			limit = {
				OR = {
					has_trait = archmage
					has_trait = true_witch
					is_a_demon_trigger = yes
				}
				any_character_artifact = {
					has_variable = blackgrimoire
				}
			}
			add = {
				value = 10
				desc = ultimate_spell_blackgrimoire
			}
		}
		if = {
			limit = {
				OR = {
					has_trait = archmage
					has_trait = true_witch
					is_a_demon_trigger = yes
				}
				any_character_artifact = {
					has_variable = catalyst
				}
			}
			add = {
				value = 10
				desc = ultimate_spell_catalyst
			}
		}
		if = {
			limit = {
				has_character_modifier = master_magic_modifier
			}
			add = {
				value = 5
				desc = ultimate_spell_master_modifier
			}
		}
	}
}
