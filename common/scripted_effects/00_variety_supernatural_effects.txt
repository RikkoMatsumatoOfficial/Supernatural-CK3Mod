st_load_intents_for_ai_effect = {
	random_list = {
		100 = {
			modifier = {
				add = 100
				OR = {

					has_trait = ambitious
					has_trait = arrogant
					has_trait = pure_blooded
				}
				NOR = {
					has_trait = shy
					has_trait = content
				}
			}
			add_character_flag = st_intent_power
		}
		100 = {
			modifier = {
				add = 100
				OR = {
					has_trait = paranoid
					has_trait = deceitful
				}
			}
			modifier = {
				add = -50
				has_trait = honest
			}
			add_character_flag = st_intent_secrecy
		}
		100 = {
			modifier = {
				add = 70
				OR = {
					has_trait = gluttonous
					has_trait = callous
					has_trait = lustful
					has_trait = lifestyle_reveler
				}
			}
			add_character_flag = st_intent_reveler
		}
	}
}

gain_st_xp1_effect = {
	if = {
		limit = {
			has_trait = supernatural_vampire
			has_trait_xp = {
				trait = supernatural_vampire
				track = vampirism
				value < 100
			}
		}
		add_trait_xp = {
			trait = supernatural_vampire
			track = vampirism
			value = 5
		}
	}
}
