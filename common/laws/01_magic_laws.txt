				add = @camp_purpose_ai_will_do_bonus_t3
			}
			### Forgiving
			if = {
				limit = { has_trait = forgiving }
				add = @camp_purpose_ai_will_do_malus_t3
			}
			# If we've picked a purpose, don't change it.
			if = {
				limit = {
					NOT = { has_realm_law = camp_purpose_wanderers }
				}
				multiply = 0
			}
		}
	}
