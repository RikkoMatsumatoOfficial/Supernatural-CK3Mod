unholy_communion = {
	icon = piety
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			location = scope:province
			has_lifestyle = learning_lifestyle
		}
	}

	weight = {
		value = 1
	}

	effect = {
		random_attending_character = {
			limit = {
				location = scope:province
				has_lifestyle = learning_lifestyle
			}
			save_scope_as = first
		}

		add_activity_log_entry = {
			key = unholy_communion
			tags = { pulse_action }
			character = scope:first

			scope:first = {
				add_piety = minor_piety_gain
			}
		}
	}
}

expertise_display = {
    icon = public
    is_valid = {
        is_current_phase_active = yes
        any_attending_character = {
            NOT = { this = scope:host }
            location = scope:province
        }
    }

	weight = {
		value = 1
		if = {
			limit = {
				scope:host = {
					prowess > 15
				}
			}
			add = 0.5
		}
	}


    effect = {
        random_attending_character = {
            limit = {
                NOT = { this = scope:host }
                location = scope:province
            }
            save_scope_as = first
        }

		if = {
			limit = {
				scope:host = {
					has_trait = supernatural_vampire
				}
			}
			add_activity_log_entry = {
				key = expertise_vampire_display
				tags = { pulse_action }
				character = scope:first

				scope:first = {
					add_opinion = {
						target = scope:root
						modifier = impressed_opinion
						opinion = 15
					}
				}
			}
		}
		else = {
			limit = {
				scope:host = {
					has_trait = supernatural_hybrid
				}
			}
			add_activity_log_entry = {
				key = expertise_hybrid_display
				tags = { pulse_action }
				character = scope:first

				scope:first = {
					add_opinion = {
						target = scope:root
						modifier = impressed_opinion
						opinion = 20
					}
				}
			}
		}
		
    }
}
