satiate_thirst_intent = {
	icon = reduce_stress_intent
	
	#ai_will_do = 1000
	ai_will_do = {
		value = 50
		if = {
			limit = {
				stress_level >= 1
			}
			add = 10
		}
		if = {
			limit = {
				stress_level >= 2
			}
			add = 10
		}
		if = {
			limit = {
				stress_level >= 3
			}
			add = 10
		}
	}
	auto_complete = yes
	
	on_invalidated = { trigger_event = activity_system.0090 }

	scripted_animation = {
		animation = personality_rational
	}
	
}

embrace_the_night_intent = {
	icon = intent_crash_wedding

	# AI behavior settings
	ai_will_do = {
		value = 50

		# Increase chance for reveler
		if = {
			limit = {
				has_trait_xp = {
					trait = lifestyle_reveler
					value > 25
				}
			}
			add = 20
		}
		if = {
			limit = {
				has_trait_xp = {
					trait = lifestyle_reveler
					value > 50
				}
			}
			add = 30
		}
		if = {
			limit = {
				has_trait_xp = {
					trait = lifestyle_reveler
					value > 75
				}
			}
			add = 40
		}

		# Increase chance for younger vampires
		if = {
			limit = {
				age < 40
			}
			add = 15
		}

		# Decrease chance for older vampires
		if = {
			limit = {
				age > 70
			}
			add = -40
		}
	}

	auto_complete = yes
	on_invalidated = { trigger_event = activity_system.0090 }

	scripted_animation = {
		animation = toast_goblet
	}
}
